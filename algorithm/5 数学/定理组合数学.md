## 康托展开
> 求 排列 `a` 是第几个排列, 因为求 `第几个`, 所以最后要 `+1`
```cpp
int a[N]={0,1,2,3,4,5};
int n=5;

int fact(int x){
    return x<1?1:x*fact(x-1);
}

int Cantor_Expansion(){ // 康托展开
    int res=0;
    for(int i=1; i<=n; i++){
        int cnt=0;
        for(int j=i+1; j<=n; j++)
            if(a[j]<a[i]) cnt++;
        res+=cnt*fact(n-i);
    }
    return res+1;
}
```

### 逆康托展开

> 给定一个 `k`, 求出 `第 k 个` 排列

```cpp
string i2s(int i){
    string res;
    stringstream ss;
    ss<<i;
    ss>>res;
    return res;
}

int fact(int x){
    return x<=1?1:x*fact(x-1);
}

string getPermutation(int n, int k) {   // 逆康托展开
    vector<int> v;
    for(int i=1; i<=n; i++) v.push_back(i); // 可选数
    k--;    // 康托展开后 +1, 是其值; 故逆康托展开需要 -1

    string res;
    for(int i=1, a, b=k; i<=n; i++){
        a = b / fact(n-i);  // 比该位小得数有 a 个
        b = b % fact(n-i);  
        res += i2s(v[a]);   // 因此选择未选数中的 a+1 个
        v.erase(v.begin()+a);
    }

    return res;
}
```
