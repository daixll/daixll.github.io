package org.dxl.utils;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jws;
import io.jsonwebtoken.JwsHeader;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.security.Keys;
import io.jsonwebtoken.security.SecureDigestAlgorithm;

import javax.crypto.SecretKey;
import java.time.Instant;
import java.util.Date;
import java.util.Map;
import java.util.UUID;

public class JwtUtils {
    private static final String SECRET = "12345678901234567890123456789012"; // 32‰∏™Â≠óËäÇÁöÑÂØÜÈí•
    private static final Integer expireTime = 12 * 60 * 60; // tokenÊúâÊïàÊúü
    private static final String ISSUER = "dxl"; // Token Á≠æÂèëËÄÖ
    private static final SecureDigestAlgorithm<SecretKey, SecretKey> ALGORITHM = Jwts.SIG.HS256; // Á≠æÂêçÁÆóÊ≥ï

    //subject: token ‰ΩøÁî®‰∏ª‰ΩìÔºåÈÄöÂ∏∏ÊòØtokenÊâÄÊúâ‰∫∫ÁöÑID
    // username Âíå userid Áî®‰∫é payload
    public static String buildJwt(String subject, Map<String, Object> claims){
        String uuid = UUID.randomUUID().toString();
        Date expireDate = Date.from(Instant.now().plusSeconds(expireTime));
        //HMAC : Hash-based Message Authentication Code
        SecretKey KEY = Keys.hmacShaKeyFor(SECRET.getBytes()); // Âü∫‰∫éÂØÜÈí•‰∫ßÁîü SecretKey
        return Jwts.builder()
                .header()
                //ËÆæÁΩÆÂ§¥ÈÉ®‰ø°ÊÅØheader
                .add("typ", "JWT")
                .add("alg", "HS256")
                .and()
                //ËÆæÁΩÆËá™ÂÆö‰πâË¥üËΩΩ‰ø°ÊÅØpayload
                .claims(claims)
                //‰ª§ÁâåID
                .id(uuid)
                // ËøáÊúüÊó∂Èó¥
                .expiration(expireDate)
                //ÁîüÊïàÊó∂Èó¥
                .notBefore(new Date())
                //Á≠æÂèëÊó∂Èó¥
                .issuedAt(new Date())
                //‰ª§ÁâåÊâÄÊúâËÄÖÔºåÈÄöÂ∏∏ÊòØÁî®Êà∑ID
                .subject(subject)
                //Á≠æÂèëËÄÖ
                .issuer(ISSUER)
                //Á≠æÂêç
                .signWith(KEY, ALGORITHM)
                .compact();
    }

    public static Jws<Claims> parseToken(String token) {
        SecretKey KEY = Keys.hmacShaKeyFor(SECRET.getBytes());
        return Jwts.parser()
                .verifyWith(KEY)
                .build()
                .parseSignedClaims(token);
    }

    public static JwsHeader parseHeader(String token) {
        return parseToken(token).getHeader();
    }

    public static Claims parsePayload(String token) {
        return parseToken(token).getPayload();
    }

    public static String getSubject(String token) {
        return parseToken(token).getPayload().getSubject();
    }

    public static boolean validateToken(String token) {
        try {
            parseToken(token);
            return true;
        } catch (Exception e) {
            return false;
        }
    }
}

package org.dxl.config;

import jakarta.annotation.Resource;
import org.springframework.context.annotation.Configuration;

import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;
import org.dxl.interceptor.LoginCheckInterceptor;

@Configuration
public class WebConfig implements WebMvcConfigurer {
    @Resource
    private LoginCheckInterceptor loginCheckInterceptor;

    @Override
    public void addInterceptors(InterceptorRegistry registry) {
        registry.addInterceptor(loginCheckInterceptor).addPathPatterns("/**").excludePathPatterns("/login");
        WebMvcConfigurer.super.addInterceptors(registry);
    }
}
