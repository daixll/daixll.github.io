const int inf = 0x3f3f3f3f;

int dis[N][N];
int m, n, t;
PII S, E;   // Ëµ∑ÁÇπÂíåÁªàÁÇπ

struct one{
    int x, y;
    int w, v;

    for(int i=1; i<=m; i++){
        string tt; cin>>tt;
        for(int j=1; j<=n; j++){
            if(tt[j-1] == '@') S = {i, j};
            if(tt[j-1] == '+') E = {i, j}, g[i][j] = 1;
            if(tt[j-1] == '*') g[i][j] = 1;
            if(tt[j-1] == '#') g[i][j] = 0;
        }
    }

    queue<one> q; q.push({S.fst, S.sed, 0, t}), dis[S.fst][S.sed] = 0;

    while(q.size()){
        auto u = q.front(); q.pop();

        if(u.x == E.fst && u.y == E.sed){
            cout<<u.w;
            return 0;
        }

        for(int i=0; i<4; i++){
            int nx = u.x + dxy[i][0];
            int ny = u.y + dxy[i][1];

                    q.push({nx, ny, u.w+1, u.v-1}), dis[nx][ny] = u.w+1;

        }
    }

    cout<<-1;
    return 0;
}
