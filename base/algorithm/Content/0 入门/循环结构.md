#include <iostream>
using namespace std;

int n, res=0x3f3f3f3f;

<details><summary><a href="https://www.luogu.com.cn/problem/P5719" target="_blank">Luogu P5719 【深基4.例3】分类平均</a> code</summary>

```cpp
#include <iostream>
using namespace std;

int n, k, m;
double A, B;

int main(){
    cin>>n>>k;
    for(int i=1; i<=n; i++)
        if(i%k==0){
            A+=i;
            m++;
        }
        else
            B+=i;
    if(m==n){
        printf("%.1lf 0.0", A/m);
        return 0;
    }
    printf("%.1lf %.1lf", A/m, B/(n-m));
    return 0;
}
```
