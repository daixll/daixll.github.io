```cpp
#include <iostream>
#include <map>
#include <queue>
using namespace std;

const int N=1;

map<int, int> pf;   // Prime Factors, 每一对指 p^c
int n;

void get_pf(int n) {
	pf.clear();
	for(int i=2; i<=n/i; i++)
		while(n%i==0){
			pf[i]++;
			n/=i;
		}
	if(n>1) pf[n]++;
}

int main(){
	cin>>n; cout<<n<<"=";
	get_pf(n);
	queue<int> q;
	for(auto [i, j]: pf)
		for(int k=1; k<=j; k++)
			q.push(i);

	while(q.size()){
		cout<<q.front(), q.pop();
		if(q.size()) cout<<"*";
	}

	return 0;
}
```
