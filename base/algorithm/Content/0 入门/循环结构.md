#include <iostream>
using namespace std;

int n, k, m;
double A, B;

int main(){
    cin>>n>>k;
    for(int i=1; i<=n; i++)
        if(i%k==0){
            A+=i;
            m++;
        }
        else
            B+=i;
    if(m==n){
        printf("%.1lf 0.0", A/m);
        return 0;

    printf("%.1lf %.1lf", A/m, B/(n-m));
