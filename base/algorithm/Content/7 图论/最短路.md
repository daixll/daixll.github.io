# Dijkstra

```cpp
int n, m; // 点、边
vector<PII> edge[100010]; // edge[x]={y,z} x->y=z
int dis[N]; // 最短路
bool st[N]; // 是否已确定最短路
// O(mlogm)
int dijkstra(int s, int e){
    // fst存dis[i] sed存i
    priority_queue<PII, vector<PII>, greater<PII>> heap;
    // 初始化
    memset(dis, 0x3f, sizeof dis);
    dis[s]=0;
    // 起点入队
    heap.push({dis[s], s});
    while(heap.size()){
        auto x=heap.top().sed; // 取最小点
        heap.pop();
        // 在出堆的时候判断、确定是否确定最短路
        if(st[x]) continue;
        st[x]=1; // 确定最短路
        // 对该点进行松弛操作
        for(auto& [y, z]: edge[x])
        if( dis[y]>dis[x]+z ){
            dis[y]=dis[x]+z;
        heap.push({dis[y], y});
    }
    }
    return dis[e];
}
```
